      }
    }
  }

  flashcardIds = new Set(flashcardIds); // Remove duplicate ids
  // retrive cards with ids from needsPractice
  let filteredCards = [];

  Array.from(flashcardIds).map((id) => {
    deck.cards.filter((card) => {
      if (card.id === id) {
        filteredCards.push(card);
      }
    });
  });

  return filteredCards;
}
